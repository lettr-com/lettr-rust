1use reqwest::header::{HeaderMap, HeaderValue, AUTHORIZATION, CONTENT_TYPE, USER_AGENT};
2use reqwest::{Method, RequestBuilder};
3
4const BASE_URL: &str = "https://app.lettr.com/api";
5
6/// Internal configuration for the Lettr HTTP client.
7#[derive(Debug, Clone)]
8pub(crate) struct Config {
9    http: reqwest::Client,
10    base_url: String,
11}
12
13impl Config {
14    /// Creates a new [`Config`] with the given API key.
15    pub fn new(api_key: &str) -> Self {
16        Self::with_client(api_key, reqwest::Client::new())
17    }
18
19    /// Creates a new [`Config`] with the given API key and a custom [`reqwest::Client`].
20    pub fn with_client(api_key: &str, client: reqwest::Client) -> Self {
21        // We'll set headers per-request since reqwest::Client doesn't allow
22        // modifying default headers after construction easily.
23        // Instead we store the client and set headers in `build`.
24        let _ = client;
25
26        let mut headers = HeaderMap::new();
27        headers.insert(
28            AUTHORIZATION,
29            HeaderValue::from_str(&format!("Bearer {api_key}"))
30                .expect("API key must be valid ASCII"),
31        );
32        headers.insert(CONTENT_TYPE, HeaderValue::from_static("application/json"));
33        headers.insert(
34            USER_AGENT,
35            HeaderValue::from_static(concat!("lettr-rust/", env!("CARGO_PKG_VERSION"))),
36        );
37
38        let http = reqwest::Client::builder()
39            .default_headers(headers)
40            .build()
41            .expect("Failed to build HTTP client");
42
43        Self {
44            http,
45            base_url: BASE_URL.to_owned(),
46        }
47    }
48
49    /// Override the base URL (useful for testing).
50    #[allow(dead_code)]
51    pub fn set_base_url(&mut self, base_url: impl Into<String>) {
52        self.base_url = base_url.into();
53    }
54
55    /// Build an HTTP request for the given method and path.
56    pub fn build(&self, method: Method, path: &str) -> RequestBuilder {
57        let url = format!("{}{path}", self.base_url);
58        self.http.request(method, url)
59    }
60
61    /// Send a built request and handle non-success status codes.
62    #[maybe_async::maybe_async]
63    pub async fn send(&self, request: RequestBuilder) -> crate::Result<reqwest::Response> {
64        let response = request.send().await?;
65        let status = response.status();
66
67        if status.is_success() {
68            Ok(response)
69        } else {
70            // Try to parse the error body
71            let body = response.text().await.unwrap_or_default();
72
73            match serde_json::from_str::<crate::error::RawErrorResponse>(&body) {
74                Ok(raw) => Err(raw.into_error()),
75                Err(_) => Err(crate::Error::Parse(format!(
76                    "HTTP {status}: {body}"
77                ))),
78            }
79        }
80    }
81}
