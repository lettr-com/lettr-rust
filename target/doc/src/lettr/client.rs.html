9/// The Lettr API client.
10///
11/// Create a client using [`Lettr::new`] with your API key, then access
12/// the various API services through the public fields.
13///
14/// # Example
15///
16/// ```rust,no_run
17/// use lettr::{Lettr, CreateEmailOptions};
18///
19/// # async fn run() -> lettr::Result<()> {
20/// let client = Lettr::new("your-api-key");
21///
22/// let email = CreateEmailOptions::new("sender@example.com", ["user@example.com"], "Hello!")
23///     .with_html("<h1>Hello World!</h1>");
24///
25/// let response = client.emails.send(email).await?;
26/// println!("Request ID: {}", response.request_id);
27/// # Ok(())
28/// # }
29/// ```
30#[derive(Clone, Debug)]
31pub struct Lettr {
32    /// Email sending, listing, and retrieval.
33    pub emails: EmailsSvc,
34    /// Domain management.
35    pub domains: DomainsSvc,
36    /// Webhook listing and retrieval.
37    pub webhooks: WebhooksSvc,
38    /// Template listing and creation.
39    pub templates: TemplatesSvc,
40
41    config: Arc<Config>,
42}
43
44impl Lettr {
45    /// Creates a new [`Lettr`] client with the given API key.
46    ///
47    /// # Panics
48    ///
49    /// Panics if the API key contains non-ASCII characters.
50    #[must_use]
51    pub fn new(api_key: &str) -> Self {
52        let config = Arc::new(Config::new(api_key));
53
54        Self {
55            emails: EmailsSvc(Arc::clone(&config)),
56            domains: DomainsSvc(Arc::clone(&config)),
57            webhooks: WebhooksSvc(Arc::clone(&config)),
58            templates: TemplatesSvc(Arc::clone(&config)),
59            config,
60        }
61    }
62
63    /// Creates a new [`Lettr`] client from the `LETTR_API_KEY` environment variable.
64    ///
65    /// # Panics
66    ///
67    /// Panics if the environment variable is not set.
68    #[must_use]
69    pub fn from_env() -> Self {
70        let api_key =
71            std::env::var("LETTR_API_KEY").expect("LETTR_API_KEY environment variable not set");
72        Self::new(&api_key)
73    }
74
75    /// Check the health of the Lettr API.
76    ///
77    /// This endpoint does not require authentication.
78    #[maybe_async::maybe_async]
79    pub async fn health(&self) -> crate::Result<HealthResponse> {
80        let request = self.config.build(reqwest::Method::GET, "/health");
81        let response = self.config.send(request).await?;
82        let body = response.json::<HealthResponse>().await?;
83        Ok(body)
84    }
85
86    /// Validate the API key and return associated team information.
87    #[maybe_async::maybe_async]
88    pub async fn auth_check(&self) -> crate::Result<AuthCheckResponse> {
89        let request = self.config.build(reqwest::Method::GET, "/auth/check");
90        let response = self.config.send(request).await?;
91        let body = response.json::<AuthCheckResponse>().await?;
92        Ok(body)
93    }
94}
